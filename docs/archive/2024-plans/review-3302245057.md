```javascript
  try {
    const result = await this.bot.postResponse(tweetId, responseText);

    if (result.success) {
      const responseTweetId = result.data?.id || result.id;
      return {
        success: true,
        responseId: responseTweetId,
        id: responseTweetId
      };
    } else {
      return {
        success: false,
        error: result.error || 'Unknown error'
      };
    }
  } catch (error) {
    return {
      success: false,
      error: error.message
    };
  }
}
```
